#define LED1 BIT0
#define LED2 BIT6

void main(void){
	WDTCTL = WDTPW |WDTHOLD;
	P1OUT = 0;
	PIDIR = LED1 + LED2;
	for (;;)
        {
        	P1OUT ^= LED1 + LED2;
        }
}

#include <msp430fr2433.h>
#define BTN BIT2
#define LED1 BIT0
#define LED2 BIT6

void main(void){
	WDTCTL = WDTPW |WDTHOLD;
	P1OUT = 0;
	PIDIR = LED1 + LED2;
        for (;;)
        {
		if(P1IN & BTN == 0)
			P1OUT= LED1 + LED2;
			P1OUT ^= LED1 + LED2;
		else 
			P1OUT &= ~(LED1 + LED2);

	}
}
